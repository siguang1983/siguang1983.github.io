    函数式编程

    纯函数式

    一、非纯净的

    let number = 1;

    const increment = () => number += 1;

    increment();  // 2


二、纯净的

    const increment = n => n + 1;

    increment(1);

    高阶函数

    把函数当做其它函数的参数来传递

let sum = (x, y) => x+y;

let calculate = (fn, x, y) => fn(x,y);

calculate(sum, 1, 2)

    柯里化

    柯里化就是将多参函数转换成单参函数

// 一个多参函数 
var add = (a, b) => a + b;
add(1, 2); // => 3

// 柯里化
var add = a => b => a+b;
add(1)(2);


Example:

    var say = function(title) {
        return function(type){
            return title + " is " + type;
        }
    }

    var sayFP = say("Functional Programming");
    var sayOther = say("Other Programming");

    sayFP("good"); // => Functional Programming is good
    sayOther("bad"); // => Other Programming is good

    compose 函数组合

    将多个函数组合成一个函数

Example:

    var compose = function(fn1, fn2) {
        return function(arg){
            return fn1(fn2(arg));
        }
    }

    var a = arg => arg + 'a';
    var b = arg => arg + 'b';

    var c = compose(a, b); // 将a,b函数进行组合
    console.log(c('c'));  // => cba

    判断数据类型

    function isDataType(type){
    var typeName = '';

    switch(toString.call(type)){
        case '[object String]':
            return 'string';
            break;

        case '[object Number]':
            if(type.toString().indexOf('.') != -1){
                return 'float';
            }
            else{
                return 'number';
            }
            break;

        case '[object Array]':
            return 'array';
            break;

        case '[object Object]':
            return 'object';
            break;

        case '[object Undefined]':
            return 'undefined';
            break;

        case '[object Null]':
            return 'null';
            break;    

        default: break;
    }

    return typeName;
}
